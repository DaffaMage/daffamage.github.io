#include <stdio.h>

int main()
{
    int damageA, damageB, armorA, armorB, hpA, hpB;
    scanf("%d %d", &hpA, &hpB);
    scanf("%d %d", &damageA, &damageB);
    scanf("%d %d", &armorA, &armorB);
    
    while(hpA>0 && hpB>0)
    {
        hpB -= damageA-armorB;
        if (hpB<=0)
        {
            break;
        }
        hpA -= damageB-armorA;
    }
    
    if (hpA>0)
    {
        printf("A %d HP Remaining", hpA);
    }
    else
    {
        printf("B %d HP Remaining", hpB);
    }

    return 0;
}
