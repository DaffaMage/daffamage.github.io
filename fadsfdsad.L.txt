#include <stdio.h>

int main()
{
    int i=0, j, N, hpA, damageA;
    scanf("%d %d", &hpA, &damageA);
    scanf("%d", &N);
    
    int hp[N], damage[N];
    
    for (j=0;j<N;j++)
    {
        printf("index %d:",j);
        scanf("%d %d", &hp[j], &damage[j]);
    }
    
    while(i<N && hpA>0)
    {
        hp[i] -= damageA;
        if (hp[i]<0)
            i++;
        for(j=0;j<N;j++)
        {
            if (hp[j]>0)
                hpA -= damage[i];
        }
    }
    
    if (i==N)
    {
        printf("MENANG");
    }
    else
    {
        printf("KALAH");
    }

    return 0;
}
